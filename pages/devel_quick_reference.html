Franklin Quick Reference

All Commands at a Glance

Installation

  • Pixi (Recommended)
  • Conda
  • Development
# Install Pixi package manager
curl -fsSL https://pixi.sh/install.sh | bash  # macOS/Linux
iwr -useb https://pixi.sh/install.ps1 | iex   # Windows PowerShell

# Students
pixi global install franklin --channel conda-forge --channel munch-group

# Educators
pixi global install franklin franklin-educator --channel conda-forge --channel munch-group

# Administrators
pixi global install franklin-admin --channel conda-forge --channel munch-group
# Students
conda install -c conda-forge -c munch-group franklin

# Educators
conda install -c conda-forge -c munch-group franklin franklin-educator

# Administrators
conda install -c conda-forge -c munch-group franklin-admin
# Clone and install in development mode
git clone https://github.com/munch-group/franklin.git
cd franklin
pip install -e .

Core Commands (Students)

Basic Operations

franklin --version              # Check version
franklin --help                 # Show help
franklin update                 # Update Franklin

Working with Exercises

franklin download               # Download exercise (interactive)
franklin jupyter                # Launch Jupyter for exercise
franklin list                   # List available exercises
franklin cleanup                # Remove old containers

Advanced Options

# Download with options
franklin download \
  --course "Data Science" \
  --exercise "Lab 1" \
  --output ~/exercises

# Launch Jupyter with resources
franklin jupyter \
  --memory 4g \
  --cpus 2 \
  --port 8889

# Cleanup old containers
franklin cleanup --days 7       # Remove containers older than 7 days
franklin cleanup --all          # Remove all Franklin containers

Educator Commands

Exercise Management

franklin exercise new           # Create new exercise
franklin exercise edit          # Edit existing exercise
franklin exercise list          # List your exercises
franklin exercise publish       # Publish to students
franklin exercise test          # Test exercise locally

Exercise Development Workflow

# 1. Create new exercise
franklin exercise new
> Course: Introduction to Python
> Exercise: Variables and Types
> Template: default

# 2. Edit exercise
cd "Variables and Types"
jupyter lab exercise.ipynb

# 3. Test locally
pixi run test-notebook

# 4. Publish for students
franklin exercise publish

Git Integration

franklin exercise clone <url>   # Clone exercise repository
franklin exercise push          # Push changes to GitLab
franklin exercise pull          # Pull latest changes
franklin exercise status        # Check git status

Administrator Commands

User Management

franklin finger <username>      # Get user information
franklin grant <role> <user>    # Grant role to user
franklin token create <user>    # Create API token
franklin password set <user>    # Set user password

Roles and Permissions

# Available roles
franklin grant student <username>
franklin grant ta <username>          # Teaching assistant
franklin grant educator <username>
franklin grant admin <username>

# Bulk operations
franklin grant student --file students.txt
franklin grant educator --course "CS101" <username>

Jupyter Magic Commands

Inside Jupyter notebooks:

# Load Franklin magic
%load_ext magic

# Install packages
%franklin numpy pandas matplotlib
%pixi_install scikit-learn tensorflow

# Install specific versions
%franklin numpy==1.24.0 pandas>=2.0

# Install from different channels
%franklin -c pytorch pytorch torchvision

Docker Commands

Container Management

# List Franklin containers
docker ps -a | grep franklin

# Stop all Franklin containers
docker stop $(docker ps -q --filter "label=franklin")

# Remove Franklin containers
docker rm $(docker ps -aq --filter "label=franklin")

# Access container shell
franklin shell                  # Interactive shell in container
franklin shell --exercise "Lab1"

Image Management

# List Franklin images
docker images | grep franklin

# Pull latest base image
docker pull registry.gitlab.com/franklin/base:latest

# Remove unused images
docker image prune

Configuration

Settings File

# Location: ~/.franklin/settings.json
{
  "default_course": "Data Science",
  "jupyter_port": 8888,
  "auto_update": true,
  "max_containers": 5,
  "default_memory": "4g",
  "default_cpus": 2
}

Environment Variables

# Debug mode
export FRANKLIN_DEBUG=1

# Custom config location
export FRANKLIN_CONFIG=/path/to/config.json

# GitLab settings
export GITLAB_URL=https://gitlab.example.com
export GITLAB_TOKEN=your-token

# Docker settings
export DOCKER_HOST=tcp://localhost:2375

Pixi Commands

In Exercise Directory

pixi install                    # Install dependencies
pixi add numpy                  # Add package
pixi remove pandas              # Remove package
pixi list                       # List installed packages
pixi run test-notebook          # Run tests
pixi shell                      # Activate environment

Global Pixi

pixi global install jupyter     # Install globally
pixi global list               # List global packages
pixi global update             # Update all packages

Git Commands for Exercises

# Check status
git status

# Save your work
git add -A
git commit -m "Complete exercise 1"

# Create your own repository
git remote add personal https://github.com/you/solutions
git push personal main

# Get updates
git pull origin main

# View history
git log --oneline

Troubleshooting Commands

Diagnostic Commands

franklin doctor                 # System diagnostics
franklin doctor --verbose       # Detailed diagnostics

# Check components
docker --version
git --version
python --version
pixi --version

Reset and Repair

# Reset Franklin configuration
franklin reset --config

# Clear cache
franklin cache clear

# Reinstall Franklin
conda remove franklin
conda install -c conda-forge -c munch-group franklin

# Fix permissions (Linux/Mac)
sudo chown -R $USER:$USER ~/.franklin

Logs and Debugging

# View logs
franklin logs                   # Recent logs
franklin logs --tail 50         # Last 50 lines
franklin logs --follow          # Follow log output

# Debug mode
FRANKLIN_DEBUG=1 franklin download

# Verbose output
franklin --verbose jupyter

Common Workflows

Student Workflow

# 1. Download exercise
franklin download

# 2. Work on exercise
franklin jupyter

# 3. Save and submit
# (Work is auto-saved in exercise folder)

Educator Workflow

# 1. Create exercise
franklin exercise new

# 2. Develop and test
cd exercise-folder
jupyter lab
pixi run test-notebook

# 3. Publish
franklin exercise publish

Administrator Workflow

# 1. Setup course
franklin grant educator prof@university.edu
franklin grant ta assistant@university.edu

# 2. Add students
franklin grant student --file roster.csv

# 3. Monitor usage
franklin stats --course "CS101"

Package Management

Adding Dependencies to Exercise

# In exercise directory
pixi add numpy pandas           # Add to pixi.toml

# Or in notebook
%franklin numpy pandas

# For specific versions
pixi add "numpy>=1.24,<2.0"

Creating Requirements File

# Export from pixi
pixi list --export > requirements.txt

# Export from conda
conda env export > environment.yml

CI/CD Integration

GitLab CI

# .gitlab-ci.yml in exercise
test:
  image: registry.gitlab.com/franklin/base:latest
  script:
    - pixi install
    - pixi run test-notebook

GitHub Actions

# .github/workflows/test.yml
name: Test Exercise
on: [push]
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - run: |
          pixi install
          pixi run test-notebook
