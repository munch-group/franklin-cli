Indexing Rows

  • Python (Pandas)
  • R (Tidyverse)
  • Julia (DataFrames.jl)
import pandas as pd

df = pd.DataFrame({
    'A': [1, 2, 3],
    'B': [4, 5, 6]
})

# Select row by position
row_pos = df.iloc[1]  # second row

# Select row by label (if index is labeled)
df.index = ['x', 'y', 'z']
row_label = df.loc['y']
library(dplyr)

df <- tibble(A = c(1, 2, 3), B = c(4, 5, 6))

# Select row by position
row_pos <- slice(df, 2)  # second row

# Select row by condition
row_condition <- filter(df, A == 2)
using DataFrames

df = DataFrame(A = [1, 2, 3], B = [4, 5, 6])

# Select row by index
row_pos = df[2, :]

# Select row by condition
row_condition = df[df.A .== 2, :]

  • Python (Pandas)
  • R (Tidyverse)
  • Julia (DataFrames.jl)
# Select a single column
col = df['A']

# Select multiple columns
cols = df[['A', 'B']]
# Select a single column
col <- df %>% select(A)

# Select multiple columns
cols <- df %>% select(A, B)
# Select a single column
col = df.A

# Select multiple columns
cols = select(df, [:A, :B])

Column and Row Assignment

  • Python (Pandas)
  • R (Tidyverse)
  • Julia (DataFrames.jl)
# Assign a new column
df['C'] = df['A'] + df['B']

# Assign values to a row by position
df.iloc[0] = [10, 20, 30]
# Add new column
mutated <- df %>% mutate(C = A + B)

# Replace first row values
df[1, ] <- list(10, 20, 30)
# Assign new column
df.C = df.A + df.B

# Replace first row
df[1, :] = (A=10, B=20, C=30)

Filtering and Masking

  • Python (Pandas)
  • R (Tidyverse)
  • Julia (DataFrames.jl)
# Boolean mask where A > 1
mask = df['A'] > 1
masked_df = df[mask]
# Logical filtering where A > 1
masked_df <- df %>% filter(A > 1)
# Filter where A > 1
masked_df = df[df.A .> 1, :]

# Long to wide
wide_df = long_df.pivot(index='A', columns='variable', values='value')
# Long to wide
wide_df <- long_df %>% pivot_wider(names_from = variable, values_from = value)
# Long to wide
wide_df = unstack(long_df, :A, :variable, :value)

Group By & Summarize

  • Python (Pandas)
  • R (Tidyverse)
  • Julia (DataFrames.jl)
# Group by A and calculate multiple summaries
summary = df.groupby('A').agg(
    B_sum=('B', 'sum'),
    C_mean=('C', 'mean')
).reset_index()
# Group by A and summarize multiple columns
summary <- df %>% 
  group_by(A) %>% 
  summarize(
    B_sum = sum(B),
    C_mean = mean(C, na.rm = TRUE)
  )
using Statistics

grouped = combine(groupby(df, :A),
    :B => sum => :B_sum,
    :C => mean => :C_mean)

# Sort by column A descending
sorted_df = df.sort_values(by='A', ascending=False)
# Sort by column A descending
sorted_df <- df %>% arrange(desc(A))
